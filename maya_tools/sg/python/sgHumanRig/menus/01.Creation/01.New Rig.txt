import sgHumanRig
import maya.cmds as cmds
import os

filePath = os.path.dirname( sgHumanRig.__file__ )  + "/data/autoRig_stdJoint.ma"

if cmds.confirmDialog( title='Confirm', message="새로운 'Scene'이 열립니다. 계속 하시겠습니까?", button=['Yes','No'], defaultButton='Yes', cancelButton='No', dismissString='No' ) == 'Yes':
    cmds.file( filePath, f=1, o=1, options="v=0;",  ignoreVersion=1, typ="mayaAscii" )
    cmds.file( rename="newRig" )
    import sgHumanRig.creation
    sgHumanRig.creation.MakeRig( 'Grp_std' )
